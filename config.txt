server {
    listen 80;
    server_name safemint.vip;
    return 301 https://safemint.vip$request_uri;
}

server {
    listen 443 ssl http2;
    listen [::]:443 ssl http2;
    server_name safemint.vip;

    root /www/wwwroot/safemint/dist;
    index index.html;

    ssl_certificate /etc/letsencrypt/live/safemint.vip/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/safemint.vip/privkey.pem;
    include /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem;

    location / {
        try_files $uri /index.html;
    }

    location ~ /\.ht {
        deny all;
    }
}


server {
    server_name app.bdcstack.com;

    location / {
        proxy_pass http://localhost:1735;   # Node.js app on port 1735
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
    }

    listen 443 ssl; # managed by Certbot
    ssl_certificate /etc/letsencrypt/live/app.bdcstack.com/fullchain.pem; # managed by Certbot

    include /etc/letsencrypt/options-ssl-nginx.conf; # managed by Certbot
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; # managed by Certbot

}
